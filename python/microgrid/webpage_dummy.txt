    <script>
        // --- System Data ---
        const hours = Array.from({length: 24}, (_, i) => i.toString().padStart(2, '0') + ":00");
        
        // Base Load Profiles
        const baseDemand = [450, 420, 400, 420, 480, 550, 650, 750, 800, 820, 800, 750, 780, 800, 850, 880, 950, 980, 850, 700, 600, 500, 450, 420];
        const extremeDemand = baseDemand.map(d => d + 60); 
        
        // Base Solar Profiles (Normalized to 3365.92 kW Base Capacity)
        const baseSolarSunny = [0,0,0,0,0,0, 50, 200, 450, 650, 800, 850, 800, 650, 450, 200, 50, 0,0,0,0,0,0,0];
        const baseSolarCloudy = [0,0,0,0,0,0, 20, 80, 150, 250, 300, 320, 300, 250, 150, 80, 20, 0,0,0,0,0,0,0];
        const BASE_SOLAR_KW = 3365.92;

        let dispatchChart = null;
        let costChart = null;

        // --- Core Simulation ---
        function runSimulation() {
            // Get inputs
            const scenario = document.getElementById('paramScenario').value;
            const capSolar = parseFloat(document.getElementById('paramSolar').value);
            const capBatt = parseFloat(document.getElementById('paramBatt').value);
            const peakLimit = parseFloat(document.getElementById('paramLimit').value);
            const rateDemand = parseFloat(document.getElementById('paramDemandRate').value);
            const rateDeg = parseFloat(document.getElementById('paramDegRate').value);

            // Update DOM text
            document.getElementById('valSolar').innerText = capSolar.toFixed(0);
            document.getElementById('valBatt').innerText = capBatt.toFixed(0);
            document.getElementById('valLimit').innerText = peakLimit;
            document.getElementById('valDemandRate').innerText = rateDemand;
            document.getElementById('valDegRate').innerText = rateDeg.toFixed(1);

            // Select Load & Solar
            const loadData = (scenario === 'extreme') ? extremeDemand : baseDemand;
            const rawSolar = (scenario === 'cloudy') ? baseSolarCloudy : baseSolarSunny;
            
            // Scale solar based on user input
            const solarFactor = capSolar / BASE_SOLAR_KW;
            const solarData = rawSolar.map(s => s * solarFactor);

            // Arrays for charting
            let gridDraw = [], solarUsedDraw = [], battDischargeDraw = [], socDraw = [];
            
            let currentSOC = 0;
            let actualPeak = 0;
            let totalDischarge = 0;

            // Physics Engine (Hourly)
            for (let i = 0; i < 24; i++) {
                let s = solarData[i];
                let d = loadData[i];
                
                // 1. Charge Battery from excess solar
                let excessSolar = Math.max(0, s - d);
                let charge = Math.min(excessSolar, capBatt - currentSOC);
                currentSOC += charge;
                
                let s_used = Math.min(d, s);
                solarUsedDraw.push(s_used);
                
                let netDemand = d - s_used;
                let discharge = 0;
                let g = netDemand;

                // 2. Discharge Battery for Peak Shaving
                if (netDemand > peakLimit) {
                    let reqDischarge = netDemand - peakLimit;
                    discharge = Math.min(reqDischarge, currentSOC); 
                    currentSOC -= discharge;
                    g = netDemand - discharge;
                    totalDischarge += discharge;
                }
                
                battDischargeDraw.push(discharge);
                gridDraw.push(g);
                socDraw.push(capBatt > 0 ? (currentSOC / capBatt) * 100 : 0);

                if (g > actualPeak) actualPeak = g;
            }

            // Economics
            const costDemand = actualPeak * rateDemand;
            const costDeg = totalDischarge * rateDeg;

            // Update Metrics
            document.getElementById('txtGridPeak').innerText = actualPeak.toFixed(0);
            document.getElementById('txtSOC').innerText = socDraw[23].toFixed(1);
            document.getElementById('txtDegCost').innerText = costDeg.toLocaleString(undefined, {maximumFractionDigits:0});
            document.getElementById('txtDemandCost').innerText = costDemand.toLocaleString(undefined, {maximumFractionDigits:0});

            // Update Dynamic Insight Box and Card Styling
            const peakCard = document.getElementById('metricCardPeak');
            const insightBox = document.getElementById('dynamicInsight');

            if (capBatt === 0) {
                peakCard.className = 'metric';
                insightBox.className = 'insight-box insight-neutral';
                insightBox.innerHTML = "<strong>System Idle:</strong> Battery capacity is set to 0. The grid absorbs the full net demand, resulting in maximum potential demand charges.";
                document.getElementById('txtGridPeak').style.color = 'var(--text-main)';
            }
            else if (actualPeak > peakLimit + 0.1) {
                peakCard.className = 'metric danger';
                insightBox.className = 'insight-box insight-warning';
                insightBox.innerHTML = `<strong>Capacity Warning:</strong> The battery depleted before the high-demand period ended. The grid spiked to ${actualPeak.toFixed(0)} kW, breaking the target threshold and incurring higher penalties. Consider increasing battery capacity or raising the target limit.`;
                document.getElementById('txtGridPeak').style.color = 'var(--c-danger)';
            } else if (totalDischarge === 0) {
                peakCard.className = 'metric';
                insightBox.className = 'insight-box insight-neutral';
                insightBox.innerHTML = "<strong>System Standby:</strong> The net demand never exceeded the Target Peak Limit. The battery remained fully charged and was not required to discharge today.";
                document.getElementById('txtGridPeak').style.color = 'var(--text-main)';
            } else {
                peakCard.className = 'metric success';
                insightBox.className = 'insight-box insight-success';
                insightBox.innerHTML = `<strong>Optimal Operation:</strong> Successful surgical peak shaving. The battery perfectly maintained the grid draw below the ${peakLimit} kW target, protecting the facility from excess demand charges.`;
                document.getElementById('txtGridPeak').style.color = 'var(--c-success)';
            }

            renderDispatchChart(gridDraw, solarUsedDraw, battDischargeDraw, loadData, socDraw, peakLimit);
            renderCostChart(costDemand, costDeg);
        }

        // --- Charting ---
        function renderDispatchChart(grid, solar, battery, demand, soc, limit) {
            const ctx = document.getElementById('dispatchChart').getContext('2d');
            if (dispatchChart) dispatchChart.destroy();

            Chart.defaults.font.family = 'Inter';
            Chart.defaults.color = '#94a3b8';

            dispatchChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: hours,
                    datasets: [
                        {
                            label: 'Solar Direct Use',
                            data: solar,
                            borderColor: 'transparent',
                            backgroundColor: 'rgba(253, 224, 71, 0.7)', // c-solar
                            fill: true,
                            yAxisID: 'y',
                            pointRadius: 0,
                            order: 4
                        },
                        {
                            label: 'Grid Power',
                            data: grid,
                            borderColor: 'transparent',
                            backgroundColor: 'rgba(226, 232, 240, 0.8)', // c-grid
                            fill: '-1',
                            yAxisID: 'y',
                            pointRadius: 0,
                            order: 3
                        },
                        {
                            label: 'Battery Discharge',
                            data: battery,
                            borderColor: 'transparent',
                            backgroundColor: 'rgba(249, 115, 22, 0.85)', // c-batt
                            fill: '-1',
                            yAxisID: 'y',
                            pointRadius: 0,
                            order: 2
                        },
                        {
                            label: 'Total Demand',
                            data: demand,
                            borderColor: '#1e293b',
                            borderWidth: 1.5,
                            borderDash: [4, 4],
                            fill: false,
                            yAxisID: 'y',
                            pointRadius: 0,
                            order: 1
                        },
                        {
                            label: 'Target Limit',
                            data: Array(24).fill(limit),
                            borderColor: '#ef4444',
                            borderWidth: 1,
                            borderDash: [2, 2],
                            fill: false,
                            yAxisID: 'y',
                            pointRadius: 0,
                            order: 0
                        },
                        {
                            label: 'SOC % (Right)',
                            data: soc,
                            borderColor: '#14b8a6',
                            borderWidth: 2,
                            fill: false,
                            yAxisID: 'y1',
                            tension: 0.3,
                            pointRadius: 0,
                            order: 5
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    interaction: { mode: 'index', intersect: false },
                    plugins: {
                        legend: {
                            position: 'top',
                            align: 'end',
                            labels: { usePointStyle: true, boxWidth: 6, font: {size: 11} }
                        },
                        tooltip: { backgroundColor: 'rgba(15, 23, 42, 0.95)', padding: 12, cornerRadius: 6, titleFont: {size: 13}, bodyFont: {size: 12} }
                    },
                    scales: {
                        y: { 
                            stacked: true, 
                            position: 'left', 
                            title: { display: true, text: 'Power (kW)', font: {weight: 500} },
                            grid: { color: '#f1f5f9', drawBorder: false }
                        },
                        y1: { 
                            position: 'right', 
                            min: 0, max: 100,
                            title: { display: true, text: 'SOC %', font: {weight: 500} },
                            grid: { display: false }
                        },
                        x: {
                            grid: { display: false },
                            ticks: { maxRotation: 0 }
                        }
                    }
                }
            });
        }

        function renderCostChart(demandCost, battCost) {
            const ctx = document.getElementById('costChart').getContext('2d');
            if (costChart) costChart.destroy();

            const total = demandCost + battCost;
            const data = total > 0 ? [demandCost, battCost] : [1];
            const bgColors = total > 0 ? ['#cbd5e1', '#f97316'] : ['#f8fafc'];
            const labels = total > 0 ? ['Demand Penalty', 'Battery Degradation'] : ['None'];

            costChart = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: labels,
                    datasets: [{
                        data: data,
                        backgroundColor: bgColors,
                        borderWidth: 2,
                        borderColor: '#ffffff',
                        hoverOffset: 4
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    cutout: '70%',
                    plugins: {
                        legend: { position: 'right', labels: { usePointStyle: true, boxWidth: 6, font: {size: 12} } },
                        tooltip: {
                            callbacks: {
                                label: function(ctx) {
                                    if (total === 0) return ' 0 THB';
                                    return ' ' + ctx.raw.toLocaleString() + ' THB';
                                }
                            }
                        }
                    }
                }
            });
        }

        function resetParams() {
            document.getElementById('paramScenario').value = 'sunny';
            document.getElementById('paramSolar').value = 3366;
            document.getElementById('paramBatt').value = 248;
            document.getElementById('paramLimit').value = 850;
            document.getElementById('paramDemandRate').value = 130;
            document.getElementById('paramDegRate').value = 3.5;
            runSimulation();
        }

        // --- Init ---
        document.querySelectorAll('input, select').forEach(el => {
            el.addEventListener('input', runSimulation);
        });

        window.onload = runSimulation;
    </script>
